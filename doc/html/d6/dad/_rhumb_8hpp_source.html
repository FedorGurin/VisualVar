101  static inline real Dtan(real x, real y) {
102  real d = x - y, tx = Math::tand(x), ty = Math::tand(y), txy = tx * ty;
103  return d ?
104  (2 * txy > -1 ? (1 + txy) * Math::tand(d) : tx - ty) /
105  (d * Math::degree()) :
106  1 + txy;
107  }
108  static inline real Datan(real x, real y) {
109  using std::atan;
110  real d = x - y, xy = x * y;
111  return d ? (2 * xy > -1 ? atan( d / (1 + xy) ) : atan(x) - atan(y)) / d :
112  1 / (1 + xy);
113  }
114  static inline real Dsin(real x, real y) {
115  using std::sin; using std::cos;
116  real d = (x - y) / 2;
117  return cos((x + y)/2) * (d ? sin(d) / d : 1);
118  }
119  static inline real Dsinh(real x, real y) {
120  using std::sinh; using std::cosh;
121  real d = (x - y) / 2;
122  return cosh((x + y) / 2) * (d ? sinh(d) / d : 1);
123  }
124  static inline real Dcosh(real x, real y) {
125  using std::sinh;
126  real d = (x - y) / 2;
127  return sinh((x + y) / 2) * (d ? sinh(d) / d : 1);
128  }
129  static inline real Dasinh(real x, real y) {
130  real d = x - y,
131  hx = Math::hypot(real(1), x), hy = Math::hypot(real(1), y);
132  return d ? Math::asinh(x*y > 0 ? d * (x + y) / (x*hy + y*hx) :
133  x*hy - y*hx) / d :
134  1 / hx;
135  }
136  static inline real Dgd(real x, real y) {
137  using std::sinh;
138  return Datan(sinh(x), sinh(y)) * Dsinh(x, y);
139  }
